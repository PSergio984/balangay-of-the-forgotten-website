"use client";

import { useRef } from "react";
import { useGSAP, gsap, ScrollTrigger } from "@/lib/gsap";

export default function AncientScrollContainer({ children }: { children: React.ReactNode }) {
  const container = useRef<HTMLDivElement>(null);
  const header = useRef<HTMLHeadingElement>(null);

  useGSAP(() => {
    gsap.from(header.current, {
      y: 30,
      opacity: 0,
      duration: 1,
      scrollTrigger: {
        trigger: container.current,
        start: "top 80%",
      }
    });
  }, { scope: container });

  return (
    <section ref={container} className="max-w-7xl mx-auto my-64 p-32 bg-[#F5E6D3] border-[12px] border-double border-midnight-blue relative shadow-2xl">
      {/* Decorative Scroll Caps */}
      <div className="absolute -top-16 -left-16 w-32 h-32 bg-midnight-blue rounded-full" />
      <div className="absolute -top-16 -right-16 w-32 h-32 bg-midnight-blue rounded-full" />
      
      <h2 ref={header} className="font-pixel text-midnight-blue text-28 mb-32 text-center">
        THE ANCIENT LOGBOOK
      </h2>
      
      <div className="font-serif-lore text-18 leading-relaxed text-midnight-blue">
        {children}
      </div>
    </section>
  );
}

- [ ] **Step 1: Create the BulletinBoard Component**
Implement the news feed with staggered "stuck note" animations.

```tsx
"use client";

import { useRef } from "react";
import { useGSAP, gsap, ScrollTrigger } from "@/lib/gsap";

const NEWS_ITEMS = [
  { id: 1, title: "New Sprite Reveal", date: "2026-05-20" },
  { id: 2, title: "Engine Alpha Build", date: "2026-05-15" },
  { id: 3, title: "Lore Update: The Void", date: "2026-05-10" },
];

export default function BulletinBoard() {
  const board = useRef<HTMLDivElement>(null);

  useGSAP(() => {
    gsap.from(".news-note", {
      scale: 0.8,
      opacity: 0,
      rotation: -5,
      stagger: 0.2,
      duration: 0.8,
      ease: "back.out(1.7)",
      scrollTrigger: {
        trigger: board.current,
        start: "top 75%",
      }
    });
  }, { scope: board });

  return (
    <section ref={board} className="max-w-7xl mx-auto my-64 p-32 bg-[#A0522D] border-8 border-midnight-blue shadow-inner relative">
      <h2 className="font-pixel text-white text-28 mb-32 text-center">DISPATCHES</h2>
      
      <div className="grid grid-cols-1 md:grid-cols-3 gap-24">
        {NEWS_ITEMS.map((item) => (
          <div key={item.id} className="news-note bg-white p-24 border-2 border-midnight-blue shadow-lg relative -rotate-2 hover:rotate-0 transition-transform cursor-pointer">
            {/* Thumbtack */}
            <div className="absolute -top-8 left-1/2 -translate-x-1/2 w-12 h-12 bg-red-600 rounded-full border border-black" />
            
            <p className="font-pixel text-12 text-gray-500 mb-8">{item.date}</p>
            <h3 className="font-pixel text-16 text-midnight-blue leading-tight">{item.title}</h3>
          </div>
        ))}
      </div>
    </section>
  );
}
```
