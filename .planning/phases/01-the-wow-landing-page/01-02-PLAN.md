Implement the layered structure for the hero. Use placeholder colors for layers until assets are ready.

```tsx
"use client";

import { useRef } from "react";
import { useGSAP, gsap, ScrollTrigger } from "@/lib/gsap";

export default function ParallaxHero() {
  const container = useRef<HTMLDivElement>(null);
  const bgRef = useRef<HTMLDivElement>(null);
  const midRef = useRef<HTMLDivElement>(null);
  const fgRef = useRef<HTMLDivElement>(null);

  useGSAP(() => {
    // Parallax logic (UI-SPEC: Background -25%, Mid -15%, Foreground -5%)
    const tl = gsap.timeline({
      scrollTrigger: {
        trigger: container.current,
        start: "top top",
        end: "bottom top",
        scrub: true,
      }
    });

    tl.to(bgRef.current, { yPercent: -25, ease: "none" }, 0);
    tl.to(midRef.current, { yPercent: -15, ease: "none" }, 0);
    tl.to(fgRef.current, { yPercent: -5, ease: "none" }, 0);
  }, { scope: container });

  return (
    <section ref={container} className="relative h-screen overflow-hidden bg-sky-mist">
      {/* Background Layer (Slowest) */}
      <div ref={bgRef} className="absolute inset-0 z-0 bg-blue-900/20" />
      
      {/* Mid Layer */}
      <div ref={midRef} className="absolute inset-0 z-10 bg-blue-700/20" />
      
      {/* Foreground Layer (Fastest) */}
      <div ref={fgRef} className="absolute inset-0 z-20 bg-blue-500/20" />

      {/* Hero Content */}
      <div className="relative z-30 h-full flex flex-col items-center justify-center text-center px-16">
        <h1 className="font-pixel text-midnight-blue text-32 mb-24 drop-shadow-pixel">
          BALANGAY OF THE FORGOTTEN
        </h1>
        <button className="bg-retro-gold border-4 border-midnight-blue p-16 font-pixel text-midnight-blue hover:scale-105 transition-transform active:translate-y-4">
          EXPLORE THE FORGOTTEN
        </button>
      </div>
    </section>
  );
}
```
